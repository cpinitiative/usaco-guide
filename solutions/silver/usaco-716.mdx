## Explanation

We want to find the number of pairs of cows for which crossing at least one road is necessary for one cow to visit the other.

Let's consider an arbitrary cow. By performing either BFS or DFS, we can determine which squares that cow can reach without crossing any roads. Then, we can determine which cows cannot be visited without crossing any roads by checking if that cow's square is reachable.

Finally, to determine all such pairs of cows, we can perform BFS or DFS for each cow and compute the number of unreachable cows, making sure to avoid double-counting pairs.

## DFS Implementation

**Time Complexity:** $\mathcal{O}(N^2)$ for DFS, and up to $\mathcal{O}(N^4)$ for computing the final answer, since there can be up to $n^2$ connected components if the roads are aggressive enough.

```cpp
#include <bits/stdc++.h>
using namespace std;

struct Road {
	int sr, sc;
	int er, ec;
};

/** to use the struct in a set, a comparator must be implemented. */
inline bool operator<(const Road &r1, const Road &r2) {
	return tie(r1.sr, r1.sc, r1.er, r1.ec) < tie(r2.sr, r2.sc, r2.er, r2.ec);
}

/** @return the 4 cardinal neighbors of a position */
vector<pair<int, int>> neighbors(int r, int c) {
	return {{r + 1, c}, {r - 1, c}, {r, c + 1}, {r, c - 1}};
}

int main() {
	freopen("countcross.in", "r", stdin);

	int side_len;
	int cow_num;
	int road_num;
	cin >> side_len >> cow_num >> road_num;

	set<Road> roads;
	for (int r = 0; r < road_num; r++) {
		int sr, sc;
		int er, ec;
		cin >> sr >> sc >> er >> ec;
		// add a road that blocks moving from both directions
		roads.insert(Road{--sr, --sc, --er, --ec});
		roads.insert(Road{er, ec, sr, sc});
	}

	vector<vector<bool>> has_cow(side_len, vector<bool>(side_len));
	for (int cow = 0; cow < cow_num; cow++) {
		int r, c;
		cin >> r >> c;
		has_cow[--r][--c] = true;
	}

	vector<vector<bool>> visited(side_len, vector<bool>(side_len));

	// returns the # of cows that a position can reach & marks them as visited
	function<int(int, int, int, int)> connected_cow_num;
	connected_cow_num = [&](int r, int c, int prev_r, int prev_c) -> int {
		// check if we're out of bounds,
		if (r < 0 || c < 0 || r >= side_len ||
		    c >= side_len
		    // someplace we've gone before,
		    || visited[r][c]
		    // or if we've just crossed a road
		    || roads.count(Road{r, c, prev_r, prev_c})) {
			return 0;
		}

		visited[r][c] = true;
		int cow_num = has_cow[r][c];
		for (const auto &[nr, nc] : neighbors(r, c)) {
			cow_num += connected_cow_num(nr, nc, r, c);
		}
		return cow_num;
	};

	vector<int> cow_components;
	for (int r = 0; r < side_len; r++) {
		for (int c = 0; c < side_len; c++) {
			if (!visited[r][c]) {
				int comp_sz = connected_cow_num(r, c, r, c);
				if (comp_sz != 0) { cow_components.push_back(comp_sz); }
			}
		}
	}

	int distant_pairs = 0;
	for (int i = 0; i < cow_components.size(); i++) {
		for (int j = i + 1; j < cow_components.size(); j++) {
			// all pairs of cows from these two components are considered
			// distant
			distant_pairs += cow_components[i] * cow_components[j];
		}
	}

	freopen("countcross.out", "w", stdout);
	cout << distant_pairs << endl;
}
```

```java
import java.io.*;
import java.util.*;

public class CountCross {
	// BeginCodeSnip{Road Class}
	private static class Road {
		int sr, sc;
		int er, ec;

		public Road(int sr, int sc, int er, int ec) {
			this.sr = sr;
			this.sc = sc;
			this.er = er;
			this.ec = ec;
		}

		@Override
		public int hashCode() {
			return Objects.hash(sr, sc, er, ec);
		}

		@Override
		public boolean equals(Object o) {
			if (!(o instanceof Road)) { return false; }
			Road r = (Road)o;
			return sr == r.sr && sc == r.sc && er == r.er && ec == r.ec;
		}
	}
	// EndCodeSnip

	private static int sideLen;
	private static Set<Road> roads = new HashSet<>();
	private static boolean[][] hasCow;
	private static boolean[][] visited;

	public static void main(String[] args) throws IOException {
		Kattio io = new Kattio("countcross");

		sideLen = io.nextInt();
		int cowNum = io.nextInt();
		int roadNum = io.nextInt();

		for (int r = 0; r < roadNum; r++) {
			int sr = io.nextInt() - 1;
			int sc = io.nextInt() - 1;
			int er = io.nextInt() - 1;
			int ec = io.nextInt() - 1;
			roads.add(new Road(sr, sc, er, ec));
			roads.add(new Road(er, ec, sr, sc));
		}

		hasCow = new boolean[sideLen][sideLen];
		for (int c = 0; c < cowNum; c++) {
			hasCow[io.nextInt() - 1][io.nextInt() - 1] = true;
		}

		visited = new boolean[sideLen][sideLen];
		List<Integer> cowComponents = new ArrayList<>();
		for (int r = 0; r < sideLen; r++) {
			for (int c = 0; c < sideLen; c++) {
				if (!visited[r][c]) {
					int compSz = connectedCowNum(r, c, r, c);
					if (compSz != 0) { cowComponents.add(compSz); }
				}
			}
		}

		int distantPairs = 0;
		for (int i = 0; i < cowComponents.size(); i++) {
			for (int j = i + 1; j < cowComponents.size(); j++) {
				// all pairs of cows from these two components are considered
				// distant
				distantPairs += cowComponents.get(i) * cowComponents.get(j);
			}
		}

		io.println(distantPairs);
		io.close();
	}

	/**
	 * @return the # of cows that a position can reach & marks them as visited
	 */
	private static int connectedCowNum(int r, int c, int prevR, int prevC) {
		// check if we're out of bounds,
		if (r < 0 || c < 0 || r >= sideLen ||
		    c >= sideLen
		    // someplace we've gone before,
		    || visited[r][c]
		    // or if we've just crossed a road
		    || roads.contains(new Road(r, c, prevR, prevC))) {
			return 0;
		}

		visited[r][c] = true;
		int cowNum = hasCow[r][c] ? 1 : 0;
		for (int[] n : neighbors(r, c)) { cowNum += connectedCowNum(n[0], n[1], r, c); }
		return cowNum;
	}

	/** @return the 4 cardinal neighbors of a position */
	private static int[][] neighbors(int r, int c) {
		return new int[][] {{r - 1, c}, {r + 1, c}, {r, c - 1}, {r, c + 1}};
	}

	// CodeSnip{Kattio}
}
```

## BFS Implementation

**Time Complexity:** Same as DFS implementation (up to $\mathcal{O}(n^4)$), see that section for details.

```
#include <fstream>
#include <cassert>
#include <cstring>
#include <queue>
#include <utility>
using namespace std;

// Maximum grid size
constexpr int MAX_N = 100;
// dx[d] and dy[d] are increments to (x, y) when travelling
// along one of the 4 standard directions; we refer to the direction
// with increment dx[d] and dy[d] as the d-th direction.
constexpr int dx[] = {1, 0, -1, 0};
constexpr int dy[] = {0, 1, 0, -1};
// can_visit[x][y][d] tracks if one can travel in the d-th direction from (x, y).
bool can_visit[MAX_N + 1][MAX_N + 1][4];
// vis[i][j] represents whether (i, j) is already visited by BFS.
bool vis[MAX_N+1][MAX_N+1];
// sz[i][j] is the number of cows in the connected component including (i, j)
// if (i, j) is where we started to BFS the connected component, and 0 otherwise.
int sz[MAX_N+1][MAX_N+1];
// cows[i][j] is number of cows at the location (i, j).
int cows[MAX_N+1][MAX_N+1];

// This bfs routine computes the amount of cows in each continous
// block as sz[sx][sy]; (sx, sy) is the start location of the BFS.
void bfs(int sx, int sy) {
    queue<pair<int, int>> q;
    // If this is visited, nothing will push and the while loop
    // below simply does not run, and sz[sx][sy] remains 0.
    if (!vis[sx][sy]) {
        vis[sx][sy] = true;
        // Accumulate the amount of cows on the starting point
        sz[sx][sy] += cows[sx][sy];
        q.push({sx, sy});
    }

    // Standard BFS boilerplate.
    while (!q.empty()) {
        auto front = q.front();
        q.pop();
        int front_x = front.first, front_y = front.second;

        // Go over the possible directions for our next location.
        for (int d = 0; d < 4; d++) {
            int next_x = front_x + dx[d], next_y = front_y + dy[d];

            // Only continue if we have not BFS'd the point and there's no
            // crossing-road blocking us.
            if (can_visit[front_x][front_y][d] && !vis[next_x][next_y])
            {
                vis[next_x][next_y] = true;
                q.push({next_x, next_y});
                // Accumulate the amount of cows at the location we're
                // going to process
                sz[sx][sy] += cows[next_x][next_y];
            }
        }
    }
}

int main() {
    ifstream fin("countcross.in");
    ofstream fout("countcross.out");
    int n;
    fin >> n;

    // Initialize all dircetions as true first, except boundary conditions handled later.
    memset(can_visit, 1, sizeof can_visit);
    // Boundary conditions: Can't go up (d=2) on first row, or go down (d=0) on last row.
    for (int j = 1; j <= n; j++)
    {
        can_visit[1][j][2] = false;
        can_visit[n][j][0] = false;
    }
    // Can't go left (d=3) on the first column, or go right (d=1) on the last column.
    for (int i = 1; i <= n; i++)
    {
        can_visit[i][1][3] = false;
        can_visit[i][n][1] = false;
    }

    int k;
    fin >> k;

    int r;
    fin >> r;
    for (int l = 1; l <= r; l++) {
        int x, y, x1, y1;
        fin >> x >> y >> x1 >> y1;

        // Figure out what d value our direction actually is.
        // Compute dx and dy values, and search those values in
        // the canoical dx[] and dy[] tables.

        // Block (x, y) to (x1, y1)
        int current_dx = x1 - x, current_dy = y1 - y;
        int d = 0;
        for (; d < 4; d++)
            if (current_dx == dx[d] && current_dy == dy[d])
                break;
        // If the search is exhausted, then something's wrong as the
        // roads are not as specified in the problem.
        assert(d<4);
        can_visit[x][y][d] = false;

        // Block (x1, y1) to (x, y); reverse the dx and dy values
        // for the oppostie direction.
        current_dx = -current_dx;
        current_dy = -current_dy;
        d = 0;
        for (; d < 4; d++)
            if (current_dx == dx[d] && current_dy == dy[d])
                break;
        assert(d<4);
        can_visit[x1][y1][d] = false;
    }

    // Convert the cow locations into the amount of cows in each
    // grid for easy accumulation in the bfs routine.
    for (int i = 1; i <= k; i++) {
        int x, y;
        fin >> x >> y;
        cows[x][y]++;
    }

    // BFS'ing from every point gets us the amount of cows in
    // each connected component, or 0 if the point belongs to another
    // connected component that is already accounted for
    // (i.e. the point does not *represent* the connected component.)
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= n; j++)
            bfs(i, j);

    // For all pairs of connected components A, B with sizes x, y,
    // respectively, there are x * y ways to pick out a cow from A
    // and a cow from B; none of these pairs can visit each other
    // without crossing a road.
    // Effectively, we simply enumerate all pairs of points, as points
    // that do not represent a connected component will have its sz[][]
    // be 0, and simply make the addition to ans ineffectual.
    int ans = 0;
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= n; j++)
            for (int i1 = 1; i1 <= n; i1++)
                for (int j1 = 1; j1 <= n; j1++)
                    if (i != i1 || j != j1)
                        ans += sz[i][j] * sz[i1][j1];
    fout << ans / 2 << endl;
    return 0;
}
```
