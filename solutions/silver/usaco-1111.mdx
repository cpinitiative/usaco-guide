```cpp
#include <algorithm>
#include <iostream>
#include <vector>

int main() {
	int n, K;
	std::cin >> n >> K;
	std::vector<int> cows(n);

	for (int &cow : cows) { std::cin >> cow; }

	std::sort(begin(cows), end(cows));

	int k = 0, d = 12;
	std::vector<int> floor(n);
	std::vector<int> jumps;

	floor[0] = cows[0] - (cows[0] % 12);
	if (floor[0] != 0) {
		k++;
		jumps.push_back(floor[0]);
	}

	for (int i = 1; i < n; i++) {
		floor[i] = cows[i] - (cows[i] % 12);
		if (floor[i] - floor[i - 1] >= 24) {
			jumps.push_back(floor[i] - floor[i - 1] - 12);
			d += 12;
			k++;
		} else if (floor[i] - floor[i - 1] == 12) {
			d += 12;
		}
	}

	std::sort(begin(jumps), end(jumps));

	int j = 0;
	while (k >= K) {
		k--;
		d += jumps[j];
		j++;
	}

	std::cout << d << '\n';
}
```

## Explanation

One thing that is clear from the problem statement is that for
Bessie to visit an ancestor, she must visit the preceding year of
the Ox (🐂) and must wait at least $12$ years to reach them. Based on
this, we can convert each ancestor's year into its $12$-year cycle
$\lfloor{\frac{year + 11}{12}}\rfloor$.

Now, we must use one jump to travel back to the earliest cycle,
leaving us with $k - 1$ jumps. With these remaining jumps, it is
optimal to skip the $k - 1$ largest gaps between consecutive
cycles in the sorted array, since **skipping a larger gap saves more time**.

We will then greedily subtract these $k - 1$ largest gaps from the
total cycle count. Finally, we multiply our result by 12 to
convert back into years, giving us the minimum time for Bessie to
visit all her ancestors and return to the present.

## Implementation

**Time Complexity:** $\mathcal{O}(N \log N)$

```cpp
#include <algorithm>
#include <iostream>
#include <vector>

int main() {
	int n, k;
	std::cin >> n >> k;
	std::vector<int> years(n);

	for (int &year : years) {
		std::cin >> year;
		(year += 11) /= 12;
	}

	years.push_back(0);  // adding current year
	std::sort(begin(years), end(years));
	years.erase(std::unique(begin(years), end(years)), end(years));

	// no jumps: Bessie must visit the furthest cycle
	int res = years.back();

	// gaps between consecutive cycles (excluding endpoints)
	std::vector<int> gaps;
	for (int i = 1; i < (int)years.size(); i++) {
		gaps.push_back(years[i] - years[i - 1] - 1);
	}

	std::sort(rbegin(gaps), rend(gaps));
	for (int i = 0; i < k - 1; i++) { res -= gaps[i]; }

	std::cout << res * 12 << '\n';
}
```

```java
			years.add((year + 11) / 12);
		}

		years.add(0);  // adding current year
		Collections.sort(years);
		years = new ArrayList<>(new LinkedHashSet<>(years));

		// no jumps: Bessie must visit the furthest cycle
		int res = years.get(years.size() - 1);

		// gaps between consecutive cycles (excluding endpoints)
		List<Integer> gaps = new ArrayList<>();
		for (int i = 1; i < years.size(); i++) {
			gaps.add(years.get(i) - years.get(i - 1) - 1);
		}

		gaps.sort(Collections.reverseOrder());
		for (int i = 0; i < k - 1 && i < gaps.size(); i++) { res -= gaps.get(i); }

		io.println(res * 12);
		io.close();
	}
```
