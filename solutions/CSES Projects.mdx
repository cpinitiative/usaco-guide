## Solution
 

[Unofficial Editorial](https://codeforces.com/blog/entry/70018).

**Time Complexity:** $\mathcal{O}(nâ‹…log n)$


```cpp

#include <bits/stdc++.h>
using namespace std;
 
int main() {
  int n;
  cin >> n;
  map<int,int> mp;
  vector<int> a(n),b(n),p(n);
  for (int i = 0; i < n; i++) {
    cin >> a[i] >> b[i] >> p[i];
    b[i]++;
    mp[a[i]], mp[b[i]];
  }
 
  int coords = 0;
  for (auto&v : mp) {
    v.second = coords++;
  }
 
  vector<vector<pair<int,int>>> v(coords);
  for (int i = 0; i < n; i++) {
    v[ mp[b[i]] ].emplace_back( mp[a[i]], p[i] );
  }
 
  vector<long long> ans(coords, 0);
  for (int i = 0; i < coords; i++) {
    if (i > 0) {
      ans[i] = ans[i-1];
    }
    for (auto p : v[i]) {
      ans[i] = max(ans[i], ans[p.first]+p.second);
    }
  }
  cout << ans[coords-1] << "\n";
}

```
