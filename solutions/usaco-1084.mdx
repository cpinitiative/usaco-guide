```java
import java.util.*;
import java.io.*;
public class Main {
	public static void main(String[] args) {
	  	Scanner sc = new Scanner(System.in);
		int O = 0;
		int E = 0;
		int x;
		int n = sc.nextInt();
		for (int i = 0; i < n; i++) {
			x = sc.nextInt();
			if (x % 2 == 0) {
				E++;
			} 
			else {
				O++;
			}
		}
		while (O > E) { 
			O = O - 2; E++; 
		}
		if (E > O + 1) {
			E = O + 1;
		}
	  	System.out.println(E + O);
  	}
}
```
