```java
import java.util.*;
import java.io.*;
public class Main {
	public static void main(String[] args) {
	  	Scanner sc = new Scanner(System.in);
		int O = 0;
		int E = 0;
		int x;
		int n = sc.nextInt();
		for (int i = 0; i < n; i++) {
			x = sc.nextInt();
			if (x % 2 == 0) {
				E++;
			} 
			else {
				O++;
			}
		}
		while (O > E) { 
			O = O - 2; E++; 
		}
		if (E > O + 1) {
			E = O + 1;
		}
	  	System.out.println(E + O);
  	}
}
```

n = int(input())
even, odd = 0, 0

cows = [int(i) for i in input().split()]

for c in cows:
	if c % 2 == 0: 
		even += 1
	else: 
		odd += 1

while odd > even:
	odd -= 2
	even += 1

if even > odd + 1: 
	even = odd + 1
print(even + odd)
