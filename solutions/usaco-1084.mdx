```java
import java.util.*;
import java.io.*;
public class Main {
	public static void main(String[] args) {
	  	Scanner sc = new Scanner(System.in);
		int O = 0;
		int E = 0;
		int x;
		int n = sc.nextInt();
		for (int i = 0; i < n; i++) {
			x = sc.nextInt();
			if (x % 2 == 0) {
				E++;
			} 
			else {
				O++;
			}
		}
		while (O > E) { 
			O = O - 2; E++; 
		}
		if (E > O + 1) {
			E = O + 1;
		}
	  	System.out.println(E + O);
  	}
}
```

n = int(input())
even, odd = 0, 0

cows = [*map(int, input().split())]

for c in cows:
    if c % 2 == 0: even += 1
    else: odd += 1

while odd > even:
    odd -= 2
    even += 1

if even > odd + 1: even = odd + 1
print(even + odd)

using namespace std;
#include<iostream>
#include<vector>

int main() {
    int even = 0;
    int odd = 0;

	int n; cin >> n;

	for(int i = 0; i < n; i++) {
		int cow; cin >> cow;
        if(cow % 2) odd++;
        else even++; 
	}

    while (even < odd) {
        even += 1;
        odd -= 2;
    }
	
    if (odd + 1 < even) even = odd + 1;

	cout << odd + even << '\n';

    return 0;
}
