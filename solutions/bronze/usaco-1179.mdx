## Explanation

**Green Squares**: Number of breeds that shares the same position
in answer and guess grid.

**Yellow Squares**: The total number of highlighted squares
$\text{(i.e., either green or yellow)}$, excluding the green
squares. Highlighted squares can be calculated by accumulating
the minimum count of each breed appearing in the answer and guess
grids.

## Implementation

**Time Complexity:** $\mathcal{O} (1)$

```cpp
#include <functional>
#include <iostream>
#include <vector>

int main() {
	char answer[3][3];
	char guess[3][3];

	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) { std::cin >> answer[i][j]; }
	}

	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) { std::cin >> guess[i][j]; }
	}

	int green_squares{0};
	int yellow_squares{0};
	std::array<int, 26> freq_answer{0};
	std::array<int, 26> freq_guess{0};

	for (int i = 0; i < 3; i++) {
		for (int j = 0; j < 3; j++) {
			green_squares += (answer[i][j] == guess[i][j]);

			freq_answer[answer[i][j] - 'A']++;
			freq_guess[guess[i][j] - 'A']++;
		}
	}

	for (int i = 0; i < 26; i++) {
		yellow_squares += std::min(freq_answer[i], freq_guess[i]);
	}

	yellow_squares -= green_squares;

	std::cout << green_squares << '\n' << yellow_squares << '\n';
}
```

```java
import java.io.*;
import java.util.*;

public class Herdle {
    public static void main(String[] args) throws Exception {
        Kattio io = new Kattio();

        char[][] answer = new char[3][3];
        char[][] guess = new char[3][3];

        for (int i = 0; i < 3; i++) {
            String row = io.next();
            for (int j = 0; j < 3; j++) {
                answer[i][j] = row.charAt(j);
            }
        }

        for (int i = 0; i < 3; i++) {
            String row = io.next();
            for (int j = 0; j < 3; j++) {
                guess[i][j] = row.charAt(j);
            }
        }

        int greenSquares = 0;
        int yellowSquares = 0;
        int[] freqAnswer = new int[26];
        int[] freqGuess = new int[26];

        for (int i = 0; i < 3; i++) {
            for (int j = 0; j < 3; j++) {
                greenSquares += (answer[i][j] == guess[i][j]) ? 1 : 0;
                freqAnswer[answer[i][j] - 'A']++;
                freqGuess[guess[i][j] - 'A']++;
            }
        }

        for (int i = 0; i < 26; i++) {
            yellowSquares += Math.min(freqAnswer[i], freqGuess[i]);
        }

        yellowSquares -= greenSquares;

        io.println(greenSquares);
        io.println(yellowSquares);
        
        io.close();
    }

    // BeginCodeSnip{Kattio}
}
```
