#include <iostream>

using namespace std;

void setIO(string s) {
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	freopen((s + ".in").c_str(), "r", stdin);
	freopen((s + ".out").c_str(), "w", stdout);
}

int N, K, A1, A2, B1, B2, res[101];  // res stores the result

int nex(int x) {
	if (A1 <= x && x <= A2)  // simulate step 1
		x = A1 + A2 - x;
	if (B1 <= x && x <= B2)  // simulate step 2
		x = B1 + B2 - x;
	return x;
}

int main() {
	setIO("swap");

	// read in input
	cin >> N >> K >> A1 >> A2 >> B1 >> B2;

	// solve for each cow
	for (int i = 1; i <= N; ++i) {
		// p = how many turns so far?
		// cur = where are we after p turns?
		int p = 1, cur = nex(i);
		while (cur != i) {  // keep going until we have found a cycle
			p++;
			cur = nex(cur);  // simulating another turn
		}
		int k = K % p;  // reduce k
		for (int j = 0; j < k; ++j) cur = nex(cur);
		res[cur] = i;  // position of cow i after k steps is cur
	}
	// print answer
	for (int i = 1; i <= N; ++i) cout << res[i] << "\n";
}

			// p = how many turns so far?
			// cur = where are we after p turns?
			int p = 1, cur = nex(i);
			while (cur != i) {  // keep going until we have found a cycle
				p++;
				cur = nex(cur);  // simulating another turn
			}
			int k = K % p;  // reduce k
			for (int j = 0; j < k; ++j) cur = nex(cur);
			res[cur] = i;  // position of cow i after k steps is cur

#include <bits/stdc++.h>
using namespace std;

int N, K;
pair<int, int> A, B;
vector<int> ori, cur;

vector<int> apply(vector<int> v) {
	reverse(begin(v) + A.first, begin(v) + A.second + 1);
	reverse(begin(v) + B.first, begin(v) + B.second + 1);
	return v;
}

int main() {
	// fastio
	cin.tie(0);
	ios::sync_with_stdio(0);
	freopen("swap.in", "r", stdin);
	freopen("swap.out", "w", stdout);

	cin >> N >> K >> A.first >> A.second >> B.first >> B.second;
	A.first--, A.second--, B.first--, B.second--;

	ori.resize(N);

	// Fill values in increasing order (ie [0, 1, 2, 3, etc])
	iota(begin(ori), end(ori), 0);

	cur = ori;
	int period = 0;

	do {
		period++;
		cur = apply(cur);
	} while (cur != ori);

	for (K %= period; K; K--) { cur = apply(cur); }

	for (int t : cur) cout << t + 1 << "\n";
}

from typing import List


def reverse_segment(cows: List[int], start: int, end: int):
	while start < end:
		cows[start], cows[end] = cows[end], cows[start]
		start += 1
		end -= 1


with open("swap.in") as read:
	n, k = map(int, read.readline().split())
	a1, a2 = map(int, read.readline().split())
	b1, b2 = map(int, read.readline().split())

cows = list(range(1, n + 1))

# apply swaps
visited = set()
visited.add(tuple(cows))
while True:
	reverse_segment(cows, a1 - 1, a2 - 1)
	reverse_segment(cows, b1 - 1, b2 - 1)
	if tuple(cows) in visited:
		break
	visited.add(tuple(cows))

cycle_len = len(visited)
k %= cycle_len
for _ in range(k):
	reverse_segment(cows, a1 - 1, a2 - 1)
	reverse_segment(cows, b1 - 1, b2 - 1)

print(*cows, sep="\n", file=open("swap.out", "w"))
