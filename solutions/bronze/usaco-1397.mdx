## Explanation

First, rearrange $S+t_i \lt c_i$ to $S \lt c_i - t_i$, and sorting it allows us to make the comparisons easier.

Then, we can check if the value located at $V$ in the array is greater than $S$. It means that following elements is possible, since it is sorted in descending order. In other words, going $V$ farms is possible when the value at $V$ is greater than $S$.

Lastly, we can print the answer according to the comparison.

## Implementation

**Time Complexity:** $\mathcal{O}(N \log N + Q)$

#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;

int main() {
	ios::sync_with_stdio(0);
    cin.tie(0);

    int n, q;
    cin >> n >> q;
    vector<int> c;
    for (int i = 0; i < n; i++) { 
        int tmp; 
        cin >> tmp; 
        c.push_back(tmp); 
    }
    for (int i = 0; i < n; i++) {
        int tmp;
        cin >> tmp;
        c[i] -= tmp;
    }

    sort(c.begin(), c.end(), greater<int>());

    for (int r = 0; r < q; r++) {
        int v, s;
        cin >> v >> s;
        if (c[v - 1] > s) {
        cout << "YES\n";
        } else {
        cout << "NO\n";
        }
    }

    return 0;
}

```java
import java.util.*;

public class MaximizingProduct {
	public static void main(String[] args) throws Exception {
		Scanner sc = new Scanner(System.in);

		int n = sc.nextInt();
		int q = sc.nextInt();

		Integer[] c = new Integer[n];
		for (int i = 0; i < n; i++) { c[i] = sc.nextInt(); }
		for (int i = 0; i < n; i++) { c[i] -= sc.nextInt(); }
		Arrays.sort(c, Collections.reverseOrder());

		for (int r = 0; r < q; r++) {
			int v = sc.nextInt();
			int s = sc.nextInt();

			if (c[v - 1] > s) {
				System.out.println("YES");
			} else {
				System.out.println("NO");
			}
		}
		sc.close();
	}
}
```
