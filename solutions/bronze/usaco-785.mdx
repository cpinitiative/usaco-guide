```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
	// file input and output
	freopen("outofplace.in", "r", stdin);
	freopen("outofplace.out", "w", stdout);

	// Read in the array size (n) and create the original and sorted arrays,
	// as well as the difference variable
	int n;
	cin >> n;
	int og[n];
	int sorted[n];
	int diff = 0;

	// Take in the values of the array as input,
	// and place them in both original and sorted
	for (int i = 0; i < n; i++) {
		cin >> og[i];
		sorted[i] = og[i];
	}

	// Actually sort the "sorted" array
	sort(sorted, sorted + n);

	// Count the number of differences in the arrays
	for (int i = 0; i < n; i++) {
		if (og[i] != sorted[i]) { diff++; }
	}

	// Print 0 if there are no differences (no swaps) or
	// diff - 1 if there are differences
	cout << max(0, diff - 1);

	return 0;
}
```

```java
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.IOException;
import java.io.PrintWriter;
import java.util.Arrays;
public class OutofPlace {
	public static void main(String[] args) throws IOException {
		// File input and output
		BufferedReader br = new BufferedReader(new FileReader("outofplace.in"));
		PrintWriter pw = new PrintWriter("outofplace.out");

		// Read in the array size (n) and create the original and sorted arrays,
		// as well as the difference variable
		int n = Integer.parseInt(br.readLine());
		int[] og = new int[n];
		int[] sort = new int[n];
		int diff = 0;

		// Take in the values of the array as input,
		// and place them in both original and sorted in the same order
		for (int i = 0; i < n; i++) {
			og[i] = Integer.parseInt(br.readLine());
			sort[i] = og[i];
		}

		// Actually sort the "sorted" array
		Arrays.sort(sort);

		// Compute the number of positions that are
		// different between the og and sorted arrays
		for (int i = 0; i < n; i++) {
			if (og[i] != sort[i]) { diff++; }
		}

		// Print 0 if there are no differences (no swaps) or diff - 1. If there
		// are differences (you need one swap for each difference, minus 1)
		pw.println(Math.max(0, diff - 1));
		pw.close();
	}
}
```

## Explanation

We start with our list of $N$ heights and make a sorted copy of it.
Then we compare the original list to the sorted list, keeping track of at which positions where they differ.

Because removing exactly one cow must make the list sorted,
those $K$ out-of-place cows form a sequence which is sorted except for one element,
so it takes exactly $K-1$ moves to put everyone back in order (and if $K \le 1$, we need zero moves).

Alternatively, we can say that the optimal solution will move exactly one cow to its proper place with each swap.
Therefore, the total number of swaps needed should be equal to the number of cows that aren't in their proper place.
However, the final swap is an exception to this rule, because it puts two cows in their proper position.
To account for this, the total number of swaps needed will be precisely one less than the total number of cows
that aren't in their proper location.

By comparing each element in the original array with the corresponding element in the sorted one
and incrementing a counter when the two elements are different, we can determine the
total number of differences between the two arrays. By subtracting 1 and ensuring the final result is greater than 0,
we produce our answer.

For example, consider the sample test case **2, 4, 7, 7, 9, 3**

To ensure that we swap the minimum number of cows as possible, we always want to swap elements of the array such that at least
one is returned to its correct position. Swapping the 2 and the 4, for example, would not be optimal, since it doesn't return
at least one cow to its proper position (in fact, this move would be actively bad, since the 2 would be moved from its correct
positioning). Since there is only 1 cow out of place, we can say that all cows after where the cow _should_ be are out of place.
In the example, cows 4-9 are out of place, because their index is one less than their index in the sorted array (where 3 comes
before all of them). The limitations of the problem force us only to swap with adjacent digits, to remain on the optimal path.
It follows that as we move the 3 to the left (closer to its intended spot), digits that fall after it will be in place,
compared to the sorted array. After the swap between 3 and 9 is made, 9 is the last element of the array, where it belongs.

```cpp
#include <algorithm>
#include <cstdio>
#include <iostream>
#include <vector>

using std::cout;
using std::endl;
using std::vector;

int main() {
	freopen("outofplace.in", "r", stdin);
	int cow_num;
	std::cin >> cow_num;
	vector<int> cows(cow_num);
	for (int &c : cows) { std::cin >> c; }

	vector<int> sorted_order(cows);
	std::sort(sorted_order.begin(), sorted_order.end());

	// The number of cows that are out of place relative to the sorted order
	int bad_num = 0;
	for (int i = 0; i < cow_num; i++) {
		if (cows[i] != sorted_order[i]) { bad_num++; }
	}

	freopen("outofplace.out", "w", stdout);
	cout << bad_num - 1 << endl;
}
```

```java
import java.io.*;
import java.util.*;

public class OutOfPlace {
	public static void main(String[] args) throws IOException {
		BufferedReader read = new BufferedReader(new FileReader("outofplace.in"));
		int cowNum = Integer.parseInt(read.readLine());
		int[] cows = new int[cowNum];
		for (int c = 0; c < cowNum; c++) {
			cows[c] = Integer.parseInt(read.readLine());
		}
		read.close();

		int[] sortedOrder = cows.clone();
		Arrays.sort(sortedOrder);

		// The number of cows that are out of place relative to the sorted order
		int badNum = 0;
		for (int i = 0; i < cowNum; i++) {
			if (cows[i] != sortedOrder[i]) { badNum++; }
		}

		PrintWriter written = new PrintWriter("outofplace.out");
		written.println(badNum - 1);
		written.close();
	}
}
```
