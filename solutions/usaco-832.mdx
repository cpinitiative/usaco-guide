## Solution 2

In this problem, there are three cases that we have to consider:
1. Cow $1$ is fixed: In this case, the solution is simple, as we just output its fixed position.
2. Cow $1$ is in the given hierarchy of cows: In this case, we want to place every cow in the hierarchy as early as possible.
3. Cow $1$ is neither fixed nor in the hierarchy: In this case, we would want to place every cow in the hierarchy as late as possible, such that cow $1$ gets the earliest possible position.

## Implementation

**Time Complexity:** $\mathcal{O}(N(M + N))$

#include <bits/stdc++.h>
using namespace std;
 
int vector_find(vector <int>& vec, int element){
    for (int i = 0; i<vec.size(); i++){
        if (vec[i] == element) return i;
    }
    return -1;
}

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

	freopen("milkorder.in", "r", stdin);
	freopen("milkorder.out", "w", stdout);

    int n, m, k, ans;
    cin>>n>>m>>k;

    vector <int> hierarchy(m);
    for (int i = 0; i<m; i++){
        cin>>hierarchy[i];
    }

    vector <vector <int>> fixed(k, vector <int> (2));
    for (int i = 0; i<k; i++){
        cin>>fixed[i][0]>>fixed[i][1];
    }

    vector <int> cows(n, 0);
    for (int i = 0; i<k; i++){
        cows[fixed[i][1] - 1] = fixed[i][0];
    }

    if (vector_find(cows, 1) != -1){
        // Case 1
        ans = vector_find(cows, 1);
    }else{
        if (vector_find(hierarchy, 1) == -1){
            // Case 2
            int i = n-1, pointer = m-1;
            while(pointer >= 0){
                if (vector_find(cows, hierarchy[pointer]) != -1){
                    i = vector_find(cows, hierarchy[pointer]) - 1;
                    pointer--;
                }else{
                    if (cows[i] == 0){
                        cows[i] = hierarchy[pointer];
                        i--; pointer--;
                    }else{
                        i--;
                    }
                }
            }
            for (int i = 0; i<n; i++){
                if (cows[i] == 0){
                    ans = i+1;
                    break;
                }
            }
        }else{
            //case 3
            int i = 0, pointer = 0;
            while(pointer < m){
                if (vector_find(cows, hierarchy[pointer]) != -1){
                    i = vector_find(cows, hierarchy[pointer]) + 1;
                    pointer++;
                }else{
                    if (cows[i] == 0){
                        cows[i] = hierarchy[pointer];
                        i++; pointer++;
                    }else{
                        i++;
                    }
                }
            }
            for (int i = 0; i<n; i++){
                if (cows[i] == 1){
                    ans = i+1;
                    break;
                }
            }
        }
    }

    cout<<ans<<endl;
    return 0;
}

file_in = open('milkorder.in')
data = file_in.read().strip().split('\n')
n, m, k = map(int, data[0].split(' '))
hierarchy = list(map(int, data[1].split(' ')))
fixed = [list(map(int, data[i + 2].split(' '))) for i in range(k)]

cows = [0 for i in range(n)]

for i in range(k):
  	cows[fixed[i][1] - 1] = fixed[i][0]

if 1 in cows:
	# Case 1
 	ans = cows.index(1) + 1
else:
	if 1 not in hierarchy:
		# Case 2
		i = n - 1
		pointer = m - 1
		while pointer >= 0:
			if hierarchy[pointer] in cows:
				i = cows.index(hierarchy[pointer]) - 1
				pointer -= 1
			else:
				if cows[i] == 0:
					cows[i] = hierarchy[pointer]
					i -= 1
					pointer -= 1
				else:
					i -= 1

		for i in range(n):
			if cows[i] == 0:
				ans = i + 1
				break
	else:
		# Case 3
		i = 0
		pointer = 0
		while pointer <= m - 1:
			if hierarchy[pointer] in cows:
				i = cows.index(hierarchy[pointer]) + 1
				pointer += 1
			else:
				if cows[i] == 0:
					cows[i] = hierarchy[pointer]
					i += 1
					pointer += 1
				else:
					i += 1
		for i in range(n):
			if cows[i] == 1:
				ans = i + 1
				break

print(ans, file=open('milkorder.out', 'w'))
