## Implementation

**Time Complexity:** $\mathcal{O}(N^2)$

<LanguageSection>

<PySection>

```python
MAX_HEIGHT = 100
MAX_DIFF = 17
	
with open('skidesign.in') as read:
	n = int(read.readline())
	elevation = [int(read.readline()) for _ in range(n)]

this_cost = float('inf')

for i in range(1, MAX_HEIGHT - MAX_DIFF):
	temp = 0
	for j in elevation:
		if j < i:
			temp += (j - i) ** 2
		elif j > i + MAX_DIFF:
			temp += (i + MAX_DIFF - j) ** 2
	this_cost = min(this_cost, temp)
    
print(this_cost, file=open('skidesign.out', 'w'))
```

```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAX_DIFF = 17;
const int MAX_HEIGHT = 100;

int elevation[1001];

int main() {
	freopen("skidesign.in", "r", stdin);
	freopen("skidesign.out", "w", stdout);

	int n;
	cin >> n;
	for (int i = 0; i < n; i++) {
		cin >> elevation[i];
	}
	
	int this_cost = INT32_MAX;
	for (int i = 0; i < MAX_HEIGHT - MAX_DIFF; i++) {
		int temp = 0;
		for (int j = 0; j < n; j++) {
			if (elevation[j] < i) {
				temp += (i - elevation[j]) * (i - elevation[j]);
			}

			if (elevation[j] > i + MAX_DIFF) {
				temp += (elevation[j] - i - MAX_DIFF) * (elevation[j] - i - MAX_DIFF);
			}
		}
		this_cost = min(this_cost, temp);
	}
	
	cout << this_cost << endl;
}
```
