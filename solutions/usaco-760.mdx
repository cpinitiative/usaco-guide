```cpp
	freopen((s+".out").c_str(),"w",stdout);

int main(){

	setIO("shuffle");

	int N; cin >> N;

	vi a(N);

	For(i, 0, N){
		cin >> a[i];
	}

	vi id(N);

	For(i, 0, N){
		cin >> id[i];
	}

	// Perform the backwards swap

	vi correct(N);

	For(i, 0, 3){
		for(int j=0; j<N; j++){
			int idx;
			for(int k=0; k<N; k++){
				if(j+1 == a[k]){
					idx = k;
					break;
				}
			}
			correct[idx] = id[j];
		}
		for(int j=0; j<N; j++){
			id[j] = correct[j];
		}
	}

	For(i, 0, N){
		cout << id[i] << endl;
	}

}
```

**Time Complexity:** $\mathcal{O}(N)$

```py
data = open('shuffle.in').read().strip().split('\n')
n = int(data[0])
moves = list(map(int, data[1].split(' ')))
order = data[2].split(' ')

for i in range(3):
    # Create a temp array to hold the order one shuffle back in time
    temp = [''] * n
    for j in range(n):
        temp[j] = order[moves[j] - 1]
    # Set this temporary array to the order of cows
    order = temp

ans = ''
for i in range(n):
    if i != n - 1:
        ans += order[i] + '\n'
    else:
        ans += order[i]

print(ans, file=open('shuffle.out', 'w'))
```
