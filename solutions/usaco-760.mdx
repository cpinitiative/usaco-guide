#include <bits/stdc++.h>

void setIO(string s) {
	ios_base::sync_with_stdio(0); cin.tie(0);
	freopen((s+".in").c_str(),"r",stdin);
	freopen((s+".out").c_str(),"w",stdout);
}

int main(){

	setIO("shuffle");

	int N; cin >> N;

	vi a(N);

	For(i, 0, N){
		cin >> a[i];
	}

	vi id(N);

	For(i, 0, N){
		cin >> id[i];
	}

	// Perform the backwards swap

	vi correct(N);

	For(i, 0, 3){
		for(int j=0; j<N; j++){
			int idx;
			for(int k=0; k<N; k++){
				if(j+1 == a[k]){
					idx = k;
					break;
				}
			}
			correct[idx] = id[j];
		}
		for(int j=0; j<N; j++){
			id[j] = correct[j];
		}
	}

	For(i, 0, N){
		cout << id[i] << endl;
	}

}
