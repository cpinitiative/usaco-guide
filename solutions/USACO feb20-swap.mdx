C++ Version of Second Solution:

```cpp
#define f first
#define s second

int N,K;
pair<int,int> A,B;
vector<int> ori, cur;
 
vector<int> apply(vector<int> v) {
	reverse(begin(v)+A.f,begin(v)+A.s+1);
	reverse(begin(v)+B.f,begin(v)+B.s+1);
	return v;
}
 
int main() {
	setIO("swap"); 
	cin >> N >> K >> A.f >> A.s >> B.f >> B.s; 
	A.f --, A.s --, B.f --, B.s --;
	ori.resize(N); iota(begin(ori),end(ori),0); cur = ori;
	int period = 0;
	do {
		period ++;
		cur = apply(cur);
	} while (cur != ori);
	for (K %= period; K; --K) cur = apply(cur);
	for (int t: cur) cout << t+1 << "\n";
}
```
