## Explanation

Each query asks: *"remove the element at the given position from what's currently in the list."*

The naive approach to do this would be to **actually** shift the elements in the array, which would be $O(n)$ per query, which simply doesn't suffice here given the constraints. We need a data structure that can answer **k-th element** queries and **deletions** efficiently.

All three solutions below do exactly that in $O(\log n)$ per operation, giving $O(n \log n)$ overall.

## Solution 1: Indexed Set

We store the **original indices** of still-present elements in an indexed set. Given a query "remove position $p$", `find_by_order(p - 1)` directly returns the original index of the $p$-th remaining element in $O(\log n)$.


<LanguageSection>
<CPPSection>

```cpp
#include <bits/stdc++.h>
using namespace std;

#include <ext/pb_ds/assoc_container.hpp>
using namespace __gnu_pbds;

template <class T>
using OrderedSet =
    tree<T, null_type, less<T>, rb_tree_tag, tree_order_statistics_node_update>;

int main() {
	int n;
	cin >> n;

	vector<int> values(n);
	for (int i = 0; i < n; i++) cin >> values[i];

	// Store original indices 0..n-1 of elements still present
	OrderedSet<int> remaining_indices;
	for (int i = 0; i < n; i++) remaining_indices.insert(i);

	for (int i = 0; i < n; i++) {
		int query_pos;
		cin >> query_pos;
		query_pos--;  // 1-indexed → 0-indexed

		// k-th smallest in remaining_indices = original index of k-th element
		int original_idx = *remaining_indices.find_by_order(query_pos);
		remaining_indices.erase(original_idx);
		cout << values[original_idx] << (i == n - 1 ? '\n' : ' ');
	}
}
```

## Solution 2: Binary Search on BIT

We maintain a BIT where each position starts as `1` (present) and gets set to `0` when removed. `prefix_sum(i)` then counts how many elements are still present in positions `1..i`.

To find the original index of the $k$-th remaining element, binary search for the smallest index where `prefix_sum(index) == k`. Each step of the binary search does a BIT query, giving $O(\log^2 n)$ per operation.

<LanguageSection>
<CPPSection>

```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;

struct BIT {
	int tree[MAXN] = {};
	int n;
	void update(int pos, int delta) {
		for (; pos <= n; pos += pos & -pos) tree[pos] += delta;
	}
	int prefix_sum(int pos) {
		int result = 0;
		for (; pos > 0; pos -= pos & -pos) result += tree[pos];
		return result;
	}
} presence;

// Binary search for smallest index where prefix_sum == target
int find_kth(int target) {
	int lo = 1, hi = presence.n;
	while (lo < hi) {
		int mid = (lo + hi) / 2;
		if (presence.prefix_sum(mid) >= target) hi = mid;
		else lo = mid + 1;
	}
	return lo;
}

int main() {
	cin.tie(0)->sync_with_stdio(0);
	cin >> presence.n;
	int n = presence.n;

	vector<int> values(n + 1);
	for (int i = 1; i <= n; i++) cin >> values[i];
	for (int i = 1; i <= n; i++) presence.update(i, 1);

	for (int q = 0; q < n; q++) {
		int query_pos;
		cin >> query_pos;
		int original_idx = find_kth(query_pos);
		cout << values[original_idx] << " \n"[q == n - 1];
		presence.update(original_idx, -1);
	}
}
```

## Solution 3

Build a segment tree where each node stores the count of present elements in its range. To find the $k$-th remaining element, [walk down the tree](https://usaco.guide/plat/segtree-ext?lang=cpp):

if the left child has $\geq k$ elements, recurse left; otherwise subtract the left child's count from $k$ and recurse right.

This finds the answer in $O(\log n)$ by descending exactly one root-to-leaf path.


<LanguageSection>
<CPPSection>

```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;

struct SegTree {
	int cnt[4 * MAXN];

	void build(int node, int lo, int hi) {
		if (lo == hi) {
			cnt[node] = 1;
			return;
		}
		int mid = (lo + hi) / 2;
		build(2 * node, lo, mid);
		build(2 * node + 1, mid + 1, hi);
		cnt[node] = cnt[2 * node] + cnt[2 * node + 1];
	}

	// Remove the k-th present element; return its original 1-indexed position
	int remove_kth(int node, int lo, int hi, int k) {
		cnt[node]--;
		if (lo == hi) return lo;
		int mid = (lo + hi) / 2;
		if (cnt[2 * node] >= k) return remove_kth(2 * node, lo, mid, k);
		else return remove_kth(2 * node + 1, mid + 1, hi, k - cnt[2 * node]);
	}
} seg;

int main() {
	cin.tie(0)->sync_with_stdio(0);
	int n;
	cin >> n;

	vector<int> values(n + 1);
	for (int i = 1; i <= n; i++) cin >> values[i];

	seg.build(1, 1, n);

	for (int q = 0; q < n; q++) {
		int query_pos;
		cin >> query_pos;
		int original_idx = seg.remove_kth(1, 1, n, query_pos);
		cout << values[original_idx] << " \n"[q == n - 1];
	}
}
```
