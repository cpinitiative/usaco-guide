## Explanation

Each query asks us to remove the element at a given position from what currently remains in the list.


The naive approach would be to shift the elements in the
array after each removal, which costs $O(n)$ per query, which
doesn't suffice here given the constraints. We need a data
structure that supports both **k-th element queries** and
**deletions** efficiently.

All three solutions below process each query in $O(\log n)$ or $O(\log^2 n)$, giving $O(n \log n)$ or $O(n\log^2 n)$ overall.


### Solution 1 - Indexed Set

Store the **original indices** of all still-present elements in an [indexed set](/gold/PURS#finding-the-k-th-element). This C++ builtin data structure can process queries of the form "remove element at position $p$", and can also access the original index at a position in $O(\log n)$ time using `find_by_order`.


<LanguageSection>
<CPPSection>

```cpp
#include <bits/stdc++.h>
using namespace std;

#include <ext/pb_ds/assoc_container.hpp>
using namespace __gnu_pbds;

template <class T>
using OrderedSet =
    tree<T, null_type, less<T>, rb_tree_tag, tree_order_statistics_node_update>;

int main() {
	int n;
	cin >> n;

	vector<int> values(n);
	for (int i = 0; i < n; i++) cin >> values[i];

	// Store original indices 0..n-1 of elements still present
	OrderedSet<int> remaining_indices;
	for (int i = 0; i < n; i++) remaining_indices.insert(i);

	for (int i = 0; i < n; i++) {
		int query_pos;
		cin >> query_pos;
		query_pos--;  // 1-indexed → 0-indexed

		// k-th smallest in remaining_indices = original index of k-th element
		int original_idx = *remaining_indices.find_by_order(query_pos);
		remaining_indices.erase(original_idx);
		cout << values[original_idx] << " \n"[i == n - 1];
	}
}
```

### Solution 2 - Binary Search on BIT

We maintain a Binary Indexed Tree (BIT) where each position is initialized to `1` (present) and gets set to `0` when removed.
The prefix sum `prefix_sum(i)` then counts how many elements are still present in positions $1$ through $i$.

To find the original index of the $k$-th remaining element, binary search for the smallest index where `prefix_sum(index) == k`. Each step of the binary search does a BIT query, giving $O(\log^2 n)$ per operation.

<LanguageSection>
<CPPSection>

```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;

struct BIT {
	int tree[MAXN] = {};
	int n;
	void update(int pos, int delta) {
		for (; pos <= n; pos += pos & -pos) tree[pos] += delta;
	}
	int prefix_sum(int pos) {
		int result = 0;
		for (; pos > 0; pos -= pos & -pos) result += tree[pos];
		return result;
	}
} presence;

// Binary search for smallest index where prefix_sum == target
int find_kth(int target) {
	int lo = 1, hi = presence.n;
	while (lo < hi) {
		int mid = (lo + hi) / 2;
		if (presence.prefix_sum(mid) >= target) hi = mid;
		else lo = mid + 1;
	}
	return lo;
}

int main() {
	cin.tie(0)->sync_with_stdio(0);
	cin >> presence.n;
	int n = presence.n;

	vector<int> values(n + 1);
	for (int i = 1; i <= n; i++) cin >> values[i];
	for (int i = 1; i <= n; i++) presence.update(i, 1);

	for (int q = 0; q < n; q++) {
		int query_pos;
		cin >> query_pos;
		int original_idx = find_kth(query_pos);
		cout << values[original_idx] << " \n"[q == n - 1];
		presence.update(original_idx, -1);
	}
}
```

### Solution 3 - Segment Tree

Build a segment tree where each node stores the count of present
elements in its range. To find the $k$-th remaining element,
[walk down the tree](/plat/segtree-ext?lang=cpp#walking-on-a-segment-tree) as follows: if the left child
contains $\geq k$ elements, recurse left; otherwise, subtract the
left child's count from $k$ and recurse right.

This finds the answer in $O(\log n)$ by descending exactly one root-to-leaf path.


<LanguageSection>
<CPPSection>

```cpp
#include <bits/stdc++.h>
using namespace std;

const int MAXN = 2e5 + 5;

struct SegTree {
	int cnt[4 * MAXN];

	void build(int node, int lo, int hi) {
		if (lo == hi) {
			cnt[node] = 1;
			return;
		}
		int mid = (lo + hi) / 2;
		build(2 * node, lo, mid);
		build(2 * node + 1, mid + 1, hi);
		cnt[node] = cnt[2 * node] + cnt[2 * node + 1];
	}

	// Remove the k-th present element; return its original 1-indexed position
	int remove_kth(int node, int lo, int hi, int k) {
		cnt[node]--;
		if (lo == hi) return lo;
		int mid = (lo + hi) / 2;
		if (cnt[2 * node] >= k) return remove_kth(2 * node, lo, mid, k);
		else return remove_kth(2 * node + 1, mid + 1, hi, k - cnt[2 * node]);
	}
} seg;

int main() {
	cin.tie(0)->sync_with_stdio(0);
	int n;
	cin >> n;

	vector<int> values(n + 1);
	for (int i = 1; i <= n; i++) cin >> values[i];

	seg.build(1, 1, n);

	for (int q = 0; q < n; q++) {
		int query_pos;
		cin >> query_pos;
		int original_idx = seg.remove_kth(1, 1, n, query_pos);
		cout << values[original_idx] << " \n"[q == n - 1];
	}
}
```
