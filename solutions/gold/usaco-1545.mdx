## Explanation

When a new party is established, quite often, it is the result of taking some cows from another party. If we think of this as each party having a set of guests attending, this would mean that in many cases, a new party forces a set to be split into 2.
Splitting a set into 2 is difficult to implement, as we need to figure out precisely how to split it. However, if we were to think about processing each query in reverse, then it turns into uniting 2 sets together, which is much easier.

A Disjoint Set Union data structure allows us to unite sets together.
In particular, once a cow decides to start hosting a party (not simply changing the type), when processed in reverse, after we've completed the query, we can simply merge that cow with their best friend to join the 2 sets together. 
Every cow in that particular party will travel to wherever the host cow goes, so we only need to merge 2 sets together.

We can use a Disjoint Set Union to represent the friendships of the cows, where each cow is a node. We add edges between a cow and their best friend depending on the situation (as described earlier), and the size() function easily lets us figure out how many cows are attending a particular cow's party.

## Implementation

**Time Complexity:** $\mathcal{O}((N+M) \alpha(N))$

#include <bits/stdc++.h>
using namespace std;

// BeginCodeSnip{Disjoint Set Union}

struct DSU {
	vector<int> e;
	DSU(int N) { e = vector<int>(N + 1, -1); }

	int get(int x) { return e[x] < 0 ? x : e[x] = get(e[x]); }

	bool same_set(int a, int b) { return get(a) == get(b); }

	int size(int x) { return -e[x]; }

	bool unite(int x, int y) {
		x = get(x), y = get(y);
		if (x == y) return false;
		swap(x, y);
		e[x] += e[y];
		e[y] = x;
		return true;
	}
};
// EndCodeSnip
int main() {
	cin.tie(0)->sync_with_stdio(0);
	int n;
	cin >> n;
	// taking in input
	vector<int> frnd(n + 1);          // each cow's best friend
	vector<char> hosted(n + 1, '0');  // what party each cow is hosting
	for (int i = 1; i <= n; i++) { cin >> frnd[i]; }
	int m;
	cin >> m;
	vector<pair<int, char>> party(m);
	for (int i = 0; i < m; i++) {
		int cow;
		cin >> cow;
		char type;
		cin >> type;
		party[i] = {cow, hosted[cow]};
		hosted[cow] = type;
	}
	// create DSU graph of friends
	DSU graph(n);
	for (int i = 1; i <= n; i++) {
		if (hosted[i] == '0') { graph.unite(i, frnd[i]); }
	}
	// first, we find out how many cows are at each type of party at the last night
	// so we can work backwards
	vector<int> typeC(m), typeO(m), typeW(m);
	for (int i = 1; i <= n; i++) {
		switch (hosted[i]) {
		case 'C':
			typeC[m - 1] += graph.size(i);
			break;
		case 'O':
			typeO[m - 1] += graph.size(i);
			break;
		case 'W':
			typeW[m - 1] += graph.size(i);
			break;
		}
	}
	// work backwards
	for (int i = m - 1; i > 0; i--) {
		// start w/ numbers from the next time
		typeC[i - 1] = typeC[i];
		typeO[i - 1] = typeO[i];
		typeW[i - 1] = typeW[i];
		int cow = party[i].first;  // which cow is hosting the new party
		// subtract
		switch (hosted[cow]) {
		case 'C':
			typeC[i - 1] -= graph.size(cow);
			break;
		case 'O':
			typeO[i - 1] -= graph.size(cow);
			break;
		case 'W':
			typeW[i - 1] -= graph.size(cow);
			break;
		}
		hosted[cow] = party[i].second;
		char prevParty =
		    hosted[cow];  // prevParty: where the new party drew customers from
		// prevParty = '0' means new party was created, so we find where this cow's best
		// friend
		if (prevParty == '0') { prevParty = hosted[graph.get(frnd[cow])]; }
		// add
		switch (prevParty) {
		case 'C':
			typeC[i - 1] += graph.size(cow);
			break;
		case 'O':
			typeO[i - 1] += graph.size(cow);
			break;
		case 'W':
			typeW[i - 1] += graph.size(cow);
			break;
		}
		// if this cow no longer hosts a party, unite
		if (hosted[cow] == '0') { graph.unite(cow, frnd[cow]); }
	}
	// print out answer
	for (int i = 0; i < m; i++) {
		cout << typeC[i] << ' ' << typeO[i] << ' ' << typeW[i] << '\n';
	}
}
