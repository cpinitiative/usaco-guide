## Explanation

We can use a Disjoint Set Union to represent the friendships of the cows, where each cow is a node and we add directed edges from each cow to their best friend.
The problem states that each cow follows their friend, so it makes sense to represent this with a DSU and unite each cow with their best friend when they aren't
hosting a party.

Since newly established parties don't suddenly disappear and at most change type, the best solution is to process each DSU query (each night) in reverse.
This way, when a cow starts to host a new party on a particular night, when done in reverse, after we've calculated the number of cows at each type of party for a
particular night, we can simply merge the host cow with their best friend.

## Implementation

**Time Complexity:** $\mathcal{O}(M \cdot \alpha(N))$

#include <bits/stdc++.h>
using namespace std;
struct DSU {
	vector<int> e;
	DSU(int N) { e = vector<int>(N + 1, -1); }

	int get(int x) { return e[x] < 0 ? x : e[x] = get(e[x]); }

	bool same_set(int a, int b) { return get(a) == get(b); }

	int size(int x) { return -e[x]; }

	bool unite(int x, int y) {
		x = get(x), y = get(y);
		if (x == y) return false;
		swap(x, y);
		e[x] += e[y];
		e[y] = x;
		return true;
	}
};
int main() {
	cin.tie(0)->sync_with_stdio(0);
	int n;
	cin >> n;
	// taking in input
	vector<int> frnd(n + 1);          // each cow's best friend
	vector<char> hosted(n + 1, '0');  // what party each cow is hosting
	for (int i = 1; i <= n; i++) { cin >> frnd[i]; }
	int m;
	cin >> m;
	vector<pair<int, char>> party(m);
	for (int i = 0; i < m; i++) {
		int cow;
		cin >> cow;
		char type;
		cin >> type;
		party[i] = {cow, hosted[cow]};
		hosted[cow] = type;
	}
	// create DSU graph of friends
	DSU graph(n);
	for (int i = 1; i <= n; i++) {
		if (hosted[i] == '0') { graph.unite(i, frnd[i]); }
	}
	// first, we find out how many cows are at each type of party at the last night
	// so we can work backwards
	vector<int> typeC(m), typeO(m), typeW(m);
	for (int i = 1; i <= n; i++) {
		switch (hosted[i]) {
		case 'C':
			typeC[m - 1] += graph.size(i);
			break;
		case 'O':
			typeO[m - 1] += graph.size(i);
			break;
		case 'W':
			typeW[m - 1] += graph.size(i);
			break;
		}
	}
	// work backwards
	for (int i = m - 1; i > 0; i--) {
		// start w/ numbers from the next time
		typeC[i - 1] = typeC[i];
		typeO[i - 1] = typeO[i];
		typeW[i - 1] = typeW[i];
		int cow = party[i].first;  // which cow is hosting the new party
		// subtract
		switch (hosted[cow]) {
		case 'C':
			typeC[i - 1] -= graph.size(cow);
			break;
		case 'O':
			typeO[i - 1] -= graph.size(cow);
			break;
		case 'W':
			typeW[i - 1] -= graph.size(cow);
			break;
		}
		hosted[cow] = party[i].second;
		char prevParty =
		    hosted[cow];  // prevParty: where the new party drew customers from
		// prevParty = '0' means new party was created, so we find where this cow's best
		// friend
		if (prevParty == '0') { prevParty = hosted[graph.get(frnd[cow])]; }
		// add
		switch (prevParty) {
		case 'C':
			typeC[i - 1] += graph.size(cow);
			break;
		case 'O':
			typeO[i - 1] += graph.size(cow);
			break;
		case 'W':
			typeW[i - 1] += graph.size(cow);
			break;
		}
		// if this cow no longer hosts a party, unite
		if (hosted[cow] == '0') { graph.unite(cow, frnd[cow]); }
	}
	// print out answer
	for (int i = 0; i < m; i++) {
		cout << typeC[i] << ' ' << typeO[i] << ' ' << typeW[i] << '\n';
	}
}
