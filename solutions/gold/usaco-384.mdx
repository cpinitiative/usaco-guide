## Explanation

First, notice that a cow can move along the edges whose elevation difference is at most $D$. So, the cells reachable from a starting point are the cells in its connected component in this graph.

Then, we sort all edges by weight and iterate through them in increasing order, similar to Kruskal’s algorithm, using a DSU to maintain connected components. As we add edges, we effectively increase the allowed $D$.

When a component first reaches size $T$, all starting points in it have found their minimum required difficulty, so we add their contribution (count multiplied by their minimum difficulty) to the total answer and reset their count to $0$ to avoid double counting.


## Implementation

**Time Complexity:** $\mathcal{O}(NM \log (NM))$

```py
import sys

sys.stdin = open("skilevel.in")
sys.stdout = open("skilevel.out", "w+")

MAXN = 500
parents = [0] * (MAXN**2)
sizes = [1] * (MAXN**2)
counts = [0] * (MAXN**2)  # the number of starting points in that component


def init(n, waypoints):
	for i in range(n):
		parents[i] = i
	for i in waypoints:
		counts[i] = 1


def find(u):
	if u == parents[u]:
		return u
	parents[u] = find(parents[u])
	return parents[u]


def unite(u, v):
	u = find(u)
	v = find(v)
	if u == v:
		return u
	if sizes[u] < sizes[v]:
		u, v = v, u
	parents[v] = u
	sizes[u] += sizes[v]
	counts[u] += counts[v]
	return u


n, m, t = map(int, input().split())
grid = [list(map(int, input().split())) for _ in range(n)]
waypointGraph = [list(map(int, input().split())) for _ in range(n)]

waypoints = [i * m + j for i in range(n) for j in range(m) if waypointGraph[i][j] == 1]

init(n * m, waypoints)

edges = []
moves = [(0, 1), (1, 0)]

# create edges with weights equivalent to elevation difference
for i in range(n):
	for j in range(m):
		for x, y in moves:
			if 0 <= i + x < n and 0 <= j + y < m:
				edges.append(
					(
						i * m + j,
						(i + x) * m + (j + y),
						abs(grid[i + x][j + y] - grid[i][j]),  # weight of the edge
					)
				)

# sort edges in increasing weights for Kruskal's
edges.sort(key=lambda x: x[2])

total = 0
for a, b, d in edges:
	root = unite(a, b)
	if sizes[root] >= t:
		total += counts[root] * d
		counts[root] = 0  # this component has now been answered

print(total)
```

Interestingly, the following code passes all test cases but won't pass if $T = 1$; you'd have to make a separate edge case for that.

```cpp
#include <bits/stdc++.h>
using namespace std;
using ll = long long;

struct Edge {
	int u, v, w;
};

struct DSU {
	vector<int> p, sz, unprocessed_starts;
	DSU(int n) : p(n), sz(n, 1), unprocessed_starts(n) { iota(p.begin(), p.end(), 0); }

	int get(int x) {
		if (p[x] != x) { p[x] = get(p[x]); }
		return p[x];
	}

	void unite(Edge e) {
		int a = get(e.u);
		int b = get(e.v);
		if (sz[a] < sz[b]) { swap(a, b); }
		if (a != b) {
			p[b] = a;
			sz[a] += sz[b];
			unprocessed_starts[a] += unprocessed_starts[b];
		}
	}
};

int main() {
	ifstream read("skilevel.in");
	int n, m, t;
	read >> n >> m >> t;
	vector<vector<int>> grid(n, vector<int>(m));
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) { read >> grid[i][j]; }
	}
	vector<vector<int>> is_start(n, vector<int>(m));
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) { read >> is_start[i][j]; }
	}

	auto to_int = [&](int r, int c) -> int { return r * m + c; };

	vector<Edge> edges;
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			// add an edge between adjacent spots
			if (i < n - 1) {
				edges.push_back(
				    {to_int(i, j), to_int(i + 1, j), abs(grid[i][j] - grid[i + 1][j])});
			}
			if (j < m - 1) {
				edges.push_back(
				    {to_int(i, j), to_int(i, j + 1), abs(grid[i][j] - grid[i][j + 1])});
			}
		}
	}

	// sort edges by weight
	sort(edges.begin(), edges.end(), [&](Edge a, Edge b) { return a.w < b.w; });

	DSU dsu(n * m);

	// initialize in dsu if this is a starting point
	for (int i = 0; i < n; i++) {
		for (int j = 0; j < m; j++) {
			if (is_start[i][j]) { dsu.unprocessed_starts[to_int(i, j)] = 1; }
		}
	}

	ll ans = 0;
	for (Edge e : edges) {
		dsu.unite(e);
		int par = dsu.get(e.u);
		if (dsu.sz[par] >= t) {
			// add number of start nodes we haven't processed * current weight
			ans += (ll)e.w * dsu.unprocessed_starts[par];
			// reset unprocessed start nodes to 0
			dsu.unprocessed_starts[par] = 0;
		}
	}

	ofstream("skilevel.out") << ans << endl;
}
```
