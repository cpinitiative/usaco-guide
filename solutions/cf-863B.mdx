```cpp
#include <bits/stdc++.h>
using namespace std;

int main() {
	ios_base::sync_with_stdio(0);
	cin.tie(0);
	int N;
	cin >> N;
	N *= 2;
	vector<int> W(N);
	for (int& x : W) cin >> x;
	sort(W.begin(), W.end());
	int min_instability = 1e9;
	for (int i = 0; i < N; ++i) {
		for (int j = i + 1; j < N; ++j) {
			vector<int> a;
			for (int k = 0; k < N; ++k)
				if (k != i && k!= j)
					a.push_back(W[k]);
			int total_instability = 0;
			for (int k = 0; k < N - 2; k += 2)
				total_instability += a[k + 1] - a[k];
			min_instability = min(min_instability, total_instability);
		}
	}
	cout << min_instability;
	return 0;
}
```

```java
import java.util.*;

public class Kayaking {

		int N = io.nextInt() * 2;

			W[x] = io.nextInt();
		}
		Arrays.sort(W);
		int minInstability = Integer.MAX_VALUE;
		for (int i = 0; i < N; i++) {
			for (int j = i + 1; j < N; j++) {
				List<Integer> track = new ArrayList<>();
				for (int k = 0; k < N; k++) {
					if ((k != i) && (k != j)) {
						track.add(W[k]);
					}
				}
				int totalInstability = 0;
				for (int k = 0; k < (N - 2); k += 2) {
					totalInstability += track.get(k + 1) - a.get(k);
				}
				minInstability = Integer.min(minInstability, totalInstability);
			}
		}
		io.println(minInstability);
		io.close();
	}

}
```
