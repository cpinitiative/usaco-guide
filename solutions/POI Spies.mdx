Same thing as Tree Matching but on a functional graph.

MLE:

```cpp
int n, spy[MX];
bool vis[MX], used[MX];
vi adj[MX];

int sum;

void dfs(int x) {
	vis[x] = 1;
	trav(t,adj[x]) if (!vis[t]) {
		dfs(t);
		if (!used[t]) used[x] = 1;
	}
	sum += used[x];
}

int main() {
	setIO(); re(n);
	FOR(i,1,n+1) {
		re(spy[i]); assert(spy[i] != i);
		adj[spy[i]].pb(i);
	}
	FOR(i,1,n+1) if (!vis[i]) {
		int x = i, y = spy[i];
		while (x != y) x = spy[x], y = spy[spy[y]];
		vi cyc = {x}; while (spy[cyc.bk] != x) cyc.pb(spy[cyc.bk]);
		trav(t,cyc) vis[t] = 1;
		vector<bool> v;
		trav(t,cyc) {
			dfs(t);
			v.pb(used[t]);
		}
		F0R(j,sz(v)) if (v[j] == 1) {
			rotate(begin(v),begin(v)+j,end(v));
			break;
		}
		int cur = 0;
		F0R(j,sz(v)) {
			if (v[j] == 0) cur ++;
			else sum += cur/2, cur = 0;
		}
		sum += cur/2;
	}
	ps(sum);
}
```

OK:

```cpp
int n, spy[MX], ST[MX], EN[MX], adj[MX];
bitset<MX> vis, used;

int sum;

void dfs(int _) {
	vi st{_};
	while (sz(st)) {
		int x = st.bk; vis[x] = 1;
		if (ST[x] == EN[x]) {
			sum += used[x];
			if (sz(st) > 1) if (!used[x]) used[st[sz(st)-2]]  = 1;
			st.pop_back(); continue;
		}
		int t = adj[ST[x]++];
		if (!vis[t]) st.pb(t);
	}
}

int main() {
	setIO(); re(n);
	FOR(i,1,n+1) {
		re(spy[i]); assert(spy[i] != i);
		ST[spy[i]] ++;
	}
	int ind = 0;
	FOR(i,1,n+1) {
		int tmp = ind; ind += ST[i];
		ST[i] = tmp; EN[i] = ind;
	}
	FOR(i,1,n+1) adj[ST[spy[i]]++] = i;
	FOR(i,1,n+1) ST[spy[i]]--;
	FOR(i,1,n+1) if (!vis[i]) {
		int x = i, y = spy[i];
		while (x != y) x = spy[x], y = spy[spy[y]];
		vi cyc = {x}; while (spy[cyc.bk] != x) cyc.pb(spy[cyc.bk]);
		trav(t,cyc) vis[t] = 1;
		vector<bool> v;
		trav(t,cyc) {
			dfs(t);
			v.pb(used[t]);
		}
		F0R(j,sz(v)) if (v[j] == 1) {
			rotate(begin(v),begin(v)+j,end(v));
			break;
		}
		int cur = 0;
		F0R(j,sz(v)) {
			if (v[j] == 0) cur ++;
			else sum += cur/2, cur = 0;
		}
		sum += cur/2;
	}
	ps(sum);
}
```
