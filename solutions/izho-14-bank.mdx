**Time Complexity:** $\mathcal{O}(M2^M)$

```cpp
int N,M;
bitset<1<<20> dp;

int main() {
	setIO(); re(N,M);
	vi a(N), b(M); re(a,b);
	vi cum{0}; trav(t,a) cum.pb(cum.bk+t);
	dp[0] = 1;
	F0R(i,1<<M) if (dp[i]) {
		int sum = 0;
		F0R(j,M) if (i&(1<<j)) sum += b[j];
		int ind = 0; while (ind+1 < sz(cum) && cum[ind+1] <= sum) ++ind;
		if (ind == sz(cum)-1) {
			ps("YES");
			exit(0);
		}
		F0R(j,M) if (!(i&(1<<j))) {
			if (sum+b[j] <= cum[ind+1]) {
				dp[i^(1<<j)] = 1;
			}
		}
	}
	ps("NO");
}
```
