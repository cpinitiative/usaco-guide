void setIO(string s) {
	ios_base::sync_with_stdio(0); cin.tie(0);
	freopen((s+".in").c_str(),"r",stdin);
	freopen((s+".out").c_str(),"w",stdout);
}

int main(){
	setIO("speeding");
	int N, M; cin >> N >> M;

	int start = 0;
	array<int,100> limit;
	For(i, 0, N){
		int length, speed; cin >> length >> speed;
		For(j, start, start+length){
			limit[j] = speed;
		}
		start += length;
	}

	start = 0;
	array<int,100> bessie;
	For(i, 0, M){
		int length, speed; cin >> length >> speed;
		For(j, start, start+length){
			bessie[j] = speed;
		}
		start += length;
	}

	int ans = 0;
	For(i, 0, 100){
		ans = max(ans, bessie[i]-limit[i]);
	}
	cout << ans << endl;
}
