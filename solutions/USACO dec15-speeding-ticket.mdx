## Implementation

```cpp
// Created by Jesse Choe - Bronze Template

#include <bits/stdc++.h>
using namespace std;

void setIO(string s) { // FastIO See General -> Input/Output
	ios_base::sync_with_stdio(0); cin.tie(0);
	freopen((s+".in").c_str(),"r",stdin);
	freopen((s+".out").c_str(),"w",stdout);
}

int main(){
	setIO("speeding");
	int N, M; cin >> N >> M;

	int start = 0;
	array<int,100> limit;
	for(int i = 0; i < N; i++){
		int length, speed; cin >> length >> speed;
		for(int j = start; i < start+length; j++){
			limit[j] = speed;
		}
		start += length;
	}

	start = 0;
	array<int,100> bessie;
	for(int i = 0; i < M; i++){
		int length, speed; cin >> length >> speed;
		for(int j = start; i < start+length; j++){
			bessie[j] = speed;
		}
		start += length;
	}

	int ans = 0;
	for(int i = 0; i < 100; i++){
		ans = max(ans, bessie[i]-limit[i]);
	}
	cout << ans << endl;
}
```
