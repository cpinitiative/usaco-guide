int main() {
	setIO("circlecross");
	str s; re(s);
	int ans = 0;
	F0R(a,sz(s)) FOR(b,a+1,sz(s)) FOR(c,b+1,sz(s)) FOR(d,c+1,sz(s))
		ans += s[a] == s[c] && s[b] == s[d];
	ps(ans);
}
