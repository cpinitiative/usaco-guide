## C++ Implementation

```cpp
int main() {
	setIO("notlast");
	int N; re(N);
	map<str,int> tot;
	F0R(i,N) {
		str a; int b; re(a,b);
		tot[a] += b;
	}
	V<pair<int,str>> cows;
	trav(t,tot) cows.pb({t.s,t.f});
	sort(all(cows));
	int ind = 0;
	if (sz(cows) == 7) {
		while (ind < sz(cows) && cows[ind].f == cows[0].f) ind ++;
	}
	if (ind < sz(cows) && (ind+1 == sz(cows) || cows[ind].f != cows[ind+1].f)) {
		ps(cows[ind].s);
		exit(0);
	}
	ps("Tie");
}
```
